                import java.awt.Color;
                import acm.graphics.GCompound;
                import acm.graphics.GOval;

                public class GEarth extends GCompound {
                    public GEarth() {
                    GOval earth = new GOval(25,25);
                    GOval moon = new GOval(10,10);
                    
                    moon.setColor(Color.decode("#dcdcdc"));
                    moon.setFillColor(Color.decode("#dcdcdc"));
                    moon.setFilled(true);
                    double xx = (getWidth() - moon.getWidth()) / 2; 
                    double yy = (getHeight() - moon.getHeight()) / 2;
                    
                    earth.setColor(Color.decode("#6b93d6"));
                    earth.setFillColor(Color.decode("#6b93d6"));
                    earth.setFilled(true);
                    double x = (getWidth() - earth.getWidth()) / 2; 
                    double y = (getHeight() - earth.getHeight()) / 2;
                    add(earth,x,y);
                    add(moon,xx,yy);
                    }
                }

            /*
            * File: SolarSystem.java
            * ----------------------
            * This program tries to simulate the solar system and revolutions of the planets around the Sun
            * Also the seasons of winter and summer
            * 
            * Author: Rares Tudor
            */
           
           
           import acm.program.*;
           import java.awt.Color;
           import acm.graphics.*;
           
           public class SolarSystem extends GraphicsProgram {
               
               private static final int MOVE = 15;
               private static final int TIME = 1;
               
               // Space
               GImage background = new GImage("C:\\Users\\Rares\\Documents\\eclipse_rares\\Image\\sky.jpg");
               // Sun
               GOval sun = new GOval(150,150);
               // Planets
               GOval mercury = new GOval(10,10);
               GOval venus = new GOval(15,15);
               GEarth earth = new GEarth();
               GOval mars = new GOval(20,20);
               GOval jupiter = new GOval(50,50);
               GSaturn saturn = new GSaturn();
               GOval uranus = new GOval(30,30);
               GOval neptune = new GOval(35,35);
               GOval pluto = new GOval(5,5);
               
               public void run() {
                   setSize(1616,862);
                   background();
                   system();
               }
               private void background() {
                   background.setSize(1616,862);
                   add(background);
               }
               
               private void system() {
                   sun();
                   mercury();
                   venus();
                   earth();
                   mars();
                   jupiter();
                   saturn();
                   uranus();
                   neptune();
                   pluto();
                   
                   // Move
                   while (true) {
                   move();
                   }
               }
               
               private void sun() {
                   sun.setFillColor(Color.decode("#FFA03C"));
                   sun.setColor(Color.decode("#FFA03C"));
                   sun.setFilled(true);
                   add(sun,getWidth() / 2 - 75, getHeight() / 2 - 75);
               }
               
               private void mercury() {
                   mercury.setColor(Color.decode("#e5e5e5"));
                   mercury.setFillColor(Color.decode("#e5e5e5"));
                   mercury.setFilled(true);
                   double x = (getWidth() - mercury.getWidth()) / 2; 
                   double y = (getHeight() - mercury.getHeight()) / 2;
                   add(mercury,x,y + 100);
               }
               
               private void venus() {
                   venus.setColor(Color.decode("#a57c1b"));
                   venus.setFillColor(Color.decode("#a57c1b"));
                   venus.setFilled(true);
                   double x = (getWidth() - venus.getWidth()) / 2; 
                   double y = (getHeight() - venus.getHeight()) / 2;
                   add(venus,x,y + 125);
               }
               
               private void earth() {
                   double x = (getWidth() - earth.getWidth()) / 2; 
                   double y = (getHeight() - earth.getHeight()) / 2;
                   add(earth,x,y + 150);
               }
               
               private void mars() {
                   mars.setColor(Color.decode("#451804"));
                   mars.setFillColor(Color.decode("#451804"));
                   mars.setFilled(true);
                   double x = (getWidth() - mars.getWidth()) / 2; 
                   double y = (getHeight() - mars.getHeight()) / 2;
                   add(mars,x,y + 175);
               }
               
               private void jupiter() {
                   jupiter.setColor(Color.decode("#bcafb2"));
                   jupiter.setFillColor(Color.decode("#bcafb2"));
                   jupiter.setFilled(true);
                   double x = (getWidth() - jupiter.getWidth()) / 2; 
                   double y = (getHeight() - jupiter.getHeight()) / 2;
                   add(jupiter,x,y + 225);
               }
               
               private void saturn() {
                   double x = (getWidth() - saturn.getWidth()) / 2; 
                   double y = (getHeight() - saturn.getHeight()) / 2;
                   add(saturn,x,y + 275);
               }
               private void uranus() {
                   uranus.setColor(Color.decode("#e1eeee"));
                   uranus.setFillColor(Color.decode("#e1eeee"));
                   uranus.setFilled(true);
                   double x = (getWidth() - uranus.getWidth()) / 2; 
                   double y = (getHeight() - uranus.getHeight()) / 2;
                   add(uranus,x,y + 325);
               }
               private void neptune() {
                   neptune.setColor(Color.decode("#9fabcf"));
                   neptune.setFillColor(Color.decode("#9fabcf"));
                   neptune.setFilled(true);
                   double x = (getWidth() - neptune.getWidth()) / 2; 
                   double y = (getHeight() - neptune.getHeight()) / 2;
                   add(neptune,x,y + 375);
               }
               private void pluto() {
                   pluto.setColor(Color.decode("#66574e"));
                   pluto.setFillColor(Color.decode("#66574e"));
                   pluto.setFilled(true);
                   double x = (getWidth() - pluto.getWidth()) / 2; 
                   double y = (getHeight() - pluto.getHeight()) / 2;
                   add(pluto,x,y + 500);
               }
               
               private void move() {
                   int x = MOVE;
                   int y = 0;
                   for (int i = 0; i < MOVE; i++) {
                       mercury.move(x, y);
                       venus.move(x * 1.25, y * 1.25);
                       earth.move(x * 1.5, y * 1.5);
                       mars.move(x * 1.8, y * 1.8);
                       jupiter.move(x * 2.5, y * 2.5);
                       saturn.move(x * 2.95, y * 2.95);
                       uranus.move(x * 3.5, y * 3.5);
                       neptune.move(x * 4, y * 4);
                       pluto.move(x * 5, y * 5);
                       
                       x -= 1;
                       y -= 1;
                       pause(TIME);
                   }
                   x = 0;
                   y = -MOVE;
                   for (int i = 0; i < MOVE; i++) {
                       mercury.move(x, y);
                       venus.move(x * 1.25, y * 1.25);
                       earth.move(x * 1.5, y * 1.5);
                       mars.move(x * 1.8, y * 1.8);
                       jupiter.move(x * 2.5, y * 2.5);
                       saturn.move(x * 2.95, y * 2.95);
                       uranus.move(x * 3.5, y * 3.5);
                       neptune.move(x * 4, y * 4);
                       pluto.move(x * 5, y * 5);
                       
                       x -= 1;
                       y += 1;
                       pause(TIME);
                   }
                   x = -MOVE;
                   y = 0;
                   for (int i = 0; i < MOVE; i++) {
                       mercury.move(x, y);
                       venus.move(x * 1.25, y * 1.25);
                       earth.move(x * 1.5, y * 1.5);
                       mars.move(x * 1.8, y * 1.8);
                       jupiter.move(x * 2.5, y * 2.5);
                       saturn.move(x * 2.95, y * 2.95);
                       uranus.move(x * 3.5, y * 3.5);
                       neptune.move(x * 4, y * 4);
                       pluto.move(x * 5, y * 5);
                       
                       x += 1;
                       y += 1;
                       pause(TIME);
                   }
                   x = 0;
                   y = MOVE;
                   for (int i = 0; i < MOVE; i++) {
                       mercury.move(x, y);
                       venus.move(x * 1.25, y * 1.25);
                       earth.move(x * 1.5, y * 1.5);
                       mars.move(x * 1.8, y * 1.8);
                       jupiter.move(x * 2.5, y * 2.5);
                       saturn.move(x * 2.95, y * 2.95);
                       uranus.move(x * 3.5, y * 3.5);
                       neptune.move(x * 4, y * 4);
                       pluto.move(x * 5, y * 5);
                       
                       x += 1;
                       y -= 1;
                       pause(TIME);
                   }
               }
            }
